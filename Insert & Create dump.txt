DROP TABLE IF EXISTS Amenities;
DROP TABLE IF EXISTS Membership;
DROP TABLE IF EXISTS Staff;
DROP TABLE IF EXISTS Service;
DROP TABLE IF EXISTS Promotion;
DROP TABLE IF EXISTS Seating;
DROP TABLE IF EXISTS Payment;
DROP TABLE IF EXISTS Booking;
DROP TABLE IF EXISTS VIPBoxes;
DROP TABLE IF EXISTS Customer;


CREATE TABLE Membership (
    MembershipID INT PRIMARY KEY AUTO_INCREMENT,
    CustomerID INT,
    MembershipType VARCHAR(50),
    StartDate DATE,
    EndDate DATE,
    FOREIGN KEY (CustomerID) REFERENCES Customer(CustomerID)
);

CREATE TABLE Payment (
    PaymentID INT PRIMARY KEY AUTO_INCREMENT,
    BookingID INT,
    Amount DECIMAL(10, 2),
    PaymentMethod VARCHAR(50),
    PaymentDate DATE,
    FOREIGN KEY (BookingID) REFERENCES Booking(BookingID)
);

CREATE TABLE Promotion (
    PromotionID INT PRIMARY KEY AUTO_INCREMENT,
    Description TEXT,
    StartDate DATE,
    EndDate DATE,
    Discount DECIMAL(5, 2)
);

CREATE TABLE Customer (
    CustomerID INT PRIMARY KEY AUTO_INCREMENT,
    Name VARCHAR(255),
    Address VARCHAR(255),
    Email VARCHAR(255),
    PhoneNo VARCHAR(15)
);

CREATE TABLE Amenities (
    AmenityID INT PRIMARY KEY AUTO_INCREMENT,
    VIPBoxID INT,
    AmenityName VARCHAR(50),
    Description TEXT,
    FOREIGN KEY (VIPBoxID) REFERENCES VIPBoxes(VIPBoxID)
);

CREATE TABLE Staff (
    StaffID INT PRIMARY KEY AUTO_INCREMENT,
    Name VARCHAR(255),
    Role VARCHAR(50),
    Contact VARCHAR(50)
);	

CREATE TABLE VIPBoxes (
    VIPBoxID INT PRIMARY KEY AUTO_INCREMENT,
    Location VARCHAR(50),
    Capacity VARCHAR(10),
    Price DECIMAL(10, 2),
    StaffID INT,
    FOREIGN KEY (StaffID) REFERENCES Staff(StaffID)
);


CREATE TABLE Booking (
    BookingID INT PRIMARY KEY AUTO_INCREMENT,
    CustomerID INT,
    VIPBoxID INT,
    BookingDate DATE,
    EventDate DATE,
    FOREIGN KEY (CustomerID) REFERENCES Customer(CustomerID),
    FOREIGN KEY (VIPBoxID) REFERENCES VIPBoxes(VIPBoxID)
);

CREATE TABLE Service (
    ServiceID INT PRIMARY KEY,
    Description VARCHAR(255),
    ServiceType VARCHAR(50),
    StaffID INT,
    FOREIGN KEY (StaffID) REFERENCES Staff(StaffID) 
);



CREATE TABLE Seating (
    SeatingID INT PRIMARY KEY AUTO_INCREMENT,
    BookingID INT,
    SeatNumber VARCHAR(10),
    Section VARCHAR(10),
    FOREIGN KEY (BookingID) REFERENCES Booking(BookingID)
);
